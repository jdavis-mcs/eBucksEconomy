<script>
    let ticketCount = 1001; // Starting ticket number

    document.getElementById('raffleForm').addEventListener('submit', function(e) {
        e.preventDefault();

        const name = document.getElementById('studentName').value;
        const grade = document.getElementById('studentGrade').value;
        const now = new Date().toLocaleString();
        
        const printArea = document.getElementById('print-area');
        
        // Generate the HTML for two copies
        const ticketHTML = `
            <div class="ticket">
                <h2>RAFFLE TICKET</h2>
                <p><strong>No: #${ticketCount}</strong></p>
                <p>Name: ${name}</p>
                <p>Grade: ${grade}</p>
                <p><small>${now}</small></p>
                <p style="margin-top:10px;">(STUDENT COPY)</p>
            </div>
            <div class="ticket">
                <h2>RAFFLE TICKET</h2>
                <p><strong>No: #${ticketCount}</strong></p>
                <p>Name: ${name}</p>
                <p>Grade: ${grade}</p>
                <p><small>${now}</small></p>
                <p style="margin-top:10px;">(BIN COPY)</p>
            </div>
        `;

        printArea.innerHTML = ticketHTML;

        // Trigger the print dialogue
        window.print();

        // Increment ticket and reset form
        ticketCount++;
        this.reset();
    });
</script>
